      useEcho('fras.alerts', '.RecognitionAlert', (payload: RecognitionAlertPayload) => {
          // Map to event
          const event = mapPayloadToEvent(payload);
          // Prepend to alerts, cap at 50
          alerts.value.unshift(event);
          if (alerts.value.length > 50) {
              alerts.value = alerts.value.slice(0, 50);
          }
          // Update today stats
          todayStats.value.recognitions++;
          if (payload.severity === 'critical') todayStats.value.critical++;
          if (payload.severity === 'warning') todayStats.value.warnings++;
          // Pulse ring on camera marker (from Plan 02)
          mapRef.value?.triggerPulse(payload.camera_id);
          // Camera recognition count
          const cam = cameras.value.find(c => c.id === payload.camera_id);
          if (cam) cam.today_recognition_count++;
          // Alert sound for critical
          if (payload.severity === 'critical') playAlertSound();
          // Highlight flash
          highlightedAlertId.value = event.id;
          setTimeout(() => { highlightedAlertId.value = null; }, 300);
      });

      const http = useHttp();
      function handleAcknowledge(event: RecognitionEvent) {
          http.submit(acknowledgeRoute.post(event), {
